27 Arch_InitIdleCpuMode(void)
28 {
29 #if LZ_CONFIG_ON_IDLE_SLEEP && \
30  ((LZ_CONFIG_AVR_SLEEP_WITH_IDLE_MODE + \
31  LZ_CONFIG_AVR_SLEEP_WITH_ADC_NOISE_REDUCTION_MODE + \
32  LZ_CONFIG_AVR_SLEEP_WITH_POWER_DOWN_MODE + \
33  LZ_CONFIG_AVR_SLEEP_WITH_POWER_SAVE_MODE + \
34  LZ_CONFIG_AVR_SLEEP_WITH_STANDBY_MODE + \
35  LZ_CONFIG_AVR_SLEEP_WITH_EXTENDED_STANDBY_MODE) > 1)
36 #error "Idle mode for architecture AVR must be unique."
37 #endif
38 
39  if (!LZ_CONFIG_ON_IDLE_SLEEP) {
40  return;
41  }
42 
43  if (LZ_CONFIG_AVR_SLEEP_WITH_ADC_NOISE_REDUCTION_MODE) {
44  SMCR = 0x02;
45  } else if (LZ_CONFIG_AVR_SLEEP_WITH_POWER_DOWN_MODE) {
46  SMCR = 0x04;
47  } else if (LZ_CONFIG_AVR_SLEEP_WITH_POWER_SAVE_MODE) {
48  SMCR = 0x06;
49  } else if (LZ_CONFIG_AVR_SLEEP_WITH_STANDBY_MODE) {
50  SMCR = 0x0c;
51  } else if (LZ_CONFIG_AVR_SLEEP_WITH_EXTENDED_STANDBY_MODE) {
52  SMCR = 0x0e;
53  } else {
54  /* Default value : LZ_CONFIG_AVR_SLEEP_WITH_IDLE_MODE */
55  SMCR = 0x00;
56  }
57 }
