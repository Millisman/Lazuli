553 void
554 Arch_SetSerialConfiguration(const Lz_SerialConfiguration * const configuration)
555 {
556  InterruptsStatus interruptsStatus;
557 
558  /* Wait all receive and transmit operations has completed */
559  /* TODO: This one doesn't work as expected */
560  /* while (!(usart->ucsr0a & UCSR0A_TXC0) || (usart->ucsr0a & UCSR0A_RXC0)); */
561 
562  /*
563  * From the ATmega328p datasheet:
564  * "For interrupt driven USART operation, the Global Interrupt Flag should be
565  * cleared (and interrupts globally disabled) when doing the initialization."
566  */
567  if (LZ_CONFIG_SERIAL_USE_INTERRUPTS) {
568  interruptsStatus = Arch_DisableInterruptsGetStatus();
569  }
570 
571  SetEnablingStatus(configuration->enableFlags);
572  SetStopBits(configuration->stopBits);
573  SetParityBit(configuration->parityBit);
574  SetSize(configuration->size);
575  SetSpeed(configuration->speed);
576 
577  if (LZ_CONFIG_SERIAL_USE_INTERRUPTS) {
578  Arch_RestoreInterruptsStatus(interruptsStatus);
579  }
